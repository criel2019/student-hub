Chapter 1: Introduction to SQL

SQL (Structured Query Language) is the standard language for interacting with relational databases. It was developed at IBM in the 1970s and is now the most widely used database language in the world.

Core RDBMS Concepts

TermMeaningExample
Table (Relation)A collection of related data in rows and columnsstudents, courses, orders
Row (Tuple / Record)A single data entry in a tableOne student's information
Column (Attribute / Field)A specific piece of data for each rowname, age, gpa
Primary Key (PK)Uniquely identifies each row; NOT NULL, UNIQUEstudent_id, order_id
Foreign Key (FK)References a PK in another table; enforces referential integrityenrollments.student_id → students.student_id
IndexData structure that speeds up queries on a columnIndex on email for fast lookup
SchemaThe structure/blueprint of a database (tables, relationships)university_db schema
NormalizationOrganizing data to reduce redundancy1NF, 2NF, 3NF, BCNF
TransactionA unit of work; all-or-nothing (ACID)Transfer money: debit + credit

Chapter 2: Creating and Managing Tables (DDL)

SQL – CREATE TABLE with All Constraint Types
-- CREATE TABLE: define structure and constraints
CREATE TABLE students (
    student_id      INT PRIMARY KEY AUTO_INCREMENT,
    first_name      VARCHAR(50)  NOT NULL,
    last_name       VARCHAR(50)  NOT NULL,
    email           VARCHAR(100) UNIQUE NOT NULL,
    age             INT          CHECK (age >= 16 AND age <= 60),
    gpa             DECIMAL(3,2) DEFAULT 0.00,
    enrollment_date DATE         DEFAULT (CURRENT_DATE),
    department      VARCHAR(50),
    phone           VARCHAR(15)  NULL
);

-- courses table
CREATE TABLE courses (
    course_id   INT PRIMARY KEY AUTO_INCREMENT,
    course_name VARCHAR(100) NOT NULL,
    credits     INT          NOT NULL CHECK (credits BETWEEN 1 AND 6),
    department  VARCHAR(50),
    instructor  VARCHAR(100),
    max_seats   INT DEFAULT 60
);

-- enrollments table with FOREIGN KEY constraints
CREATE TABLE enrollments (
    enrollment_id INT PRIMARY KEY AUTO_INCREMENT,
    student_id    INT NOT NULL,
    course_id     INT NOT NULL,
    grade         VARCHAR(2)  DEFAULT NULL,
    semester      VARCHAR(20) NOT NULL,
    enrolled_at   TIMESTAMP   DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (student_id) REFERENCES students(student_id)
        ON DELETE CASCADE     -- delete enrollment if student deleted
        ON UPDATE CASCADE,    -- update if student_id changes
    FOREIGN KEY (course_id)  REFERENCES courses(course_id)
        ON DELETE RESTRICT,   -- prevent deleting a course with enrollments
    UNIQUE (student_id, course_id, semester)  -- composite unique
);

-- CREATE TABLE LIKE — copy structure without data
CREATE TABLE students_backup LIKE students;

-- CREATE TABLE AS SELECT — copy structure AND data
CREATE TABLE cs_students AS
    SELECT * FROM students WHERE department = 'Computer Science';
SQL – ALTER TABLE
-- Add a new column
ALTER TABLE students ADD COLUMN address VARCHAR(200);
ALTER TABLE students ADD COLUMN birth_year INT AFTER age;

-- Modify column type/constraints
ALTER TABLE students MODIFY COLUMN gpa DECIMAL(4,2);          -- MySQL
ALTER TABLE students ALTER COLUMN gpa TYPE DECIMAL(4,2);      -- PostgreSQL

-- Rename a column (MySQL 8.0+)
ALTER TABLE students RENAME COLUMN phone TO mobile;

-- Drop a column
ALTER TABLE students DROP COLUMN address;

-- Add a constraint after table creation
ALTER TABLE students ADD CONSTRAINT chk_gpa CHECK (gpa >= 0 AND gpa <= 10);
ALTER TABLE students ADD CONSTRAINT uq_email UNIQUE (email);

-- Drop a constraint
ALTER TABLE students DROP INDEX uq_email;     -- MySQL
ALTER TABLE students DROP CONSTRAINT uq_email; -- PostgreSQL, MS SQL

-- Rename the entire table
RENAME TABLE students TO university_students;  -- MySQL
ALTER TABLE students RENAME TO university_students;  -- PostgreSQL
SQL – DROP and TRUNCATE
-- DROP TABLE: remove table and all its data permanently
DROP TABLE IF EXISTS students_backup;
DROP TABLE IF EXISTS temp_results;

-- TRUNCATE: delete all rows but keep the table structure
-- Faster than DELETE, resets AUTO_INCREMENT
TRUNCATE TABLE students_backup;

-- DROP DATABASE (careful!)
DROP DATABASE IF EXISTS test_db;

-- VIEW: virtual table based on a query
CREATE VIEW cs_student_view AS
    SELECT student_id, first_name, last_name, gpa
    FROM students
    WHERE department = 'Computer Science';

-- Query a view just like a table
SELECT * FROM cs_student_view WHERE gpa > 8.0;

-- Drop a view
DROP VIEW IF EXISTS cs_student_view;

Chapter 3: Inserting, Updating, and Deleting Data (DML)

SQL – INSERT
-- Insert a single row (all columns)
INSERT INTO students VALUES (
    NULL, 'Rahul', 'Sharma', 'rahul@example.com', 20, 8.5, '2023-07-01', 'Computer Science', NULL
);

-- Insert with specified columns (recommended — order-independent)
INSERT INTO students (first_name, last_name, email, age, gpa, department)
VALUES ('Priya', 'Patel', 'priya@example.com', 21, 9.2, 'Electronics');

-- Insert multiple rows in one statement (more efficient)
INSERT INTO students (first_name, last_name, email, age, gpa, department) VALUES
    ('Amit',   'Kumar',  'amit@example.com',   19, 7.8, 'Mechanical'),
    ('Sunita', 'Singh',  'sunita@example.com',  22, 8.9, 'Computer Science'),
    ('Wei',    'Zhang',  'wei@example.com',     20, 9.5, 'Mathematics'),
    ('Aisha',  'Khan',   'aisha@example.com',   20, 9.1, 'Computer Science'),
    ('Carlos', 'Rivera', 'carlos@example.com',  23, 8.3, 'Electronics');

-- INSERT ... SELECT: copy data from another table
INSERT INTO cs_students (first_name, last_name, email, gpa)
    SELECT first_name, last_name, email, gpa
    FROM students
    WHERE department = 'Computer Science';

-- REPLACE INTO: delete existing row and insert (MySQL-specific)
-- Useful when you want upsert behavior
REPLACE INTO students (student_id, first_name, last_name, email, age, gpa, department)
VALUES (1, 'Rahul', 'Sharma', 'rahul.sharma@example.com', 21, 8.7, 'Computer Science');

-- INSERT ... ON DUPLICATE KEY UPDATE (MySQL upsert)
INSERT INTO students (student_id, email, gpa)
VALUES (1, 'new_email@example.com', 9.0)
ON DUPLICATE KEY UPDATE
    email = VALUES(email),
    gpa   = VALUES(gpa);
SQL – UPDATE
-- Update a single column for one row
UPDATE students SET gpa = 9.0 WHERE student_id = 1;

-- Update multiple columns
UPDATE students
SET gpa = 8.8, department = 'Computer Science'
WHERE first_name = 'Amit' AND last_name = 'Kumar';

-- Update using arithmetic (give everyone in CS a 0.1 GPA boost)
UPDATE students
SET gpa = ROUND(gpa + 0.1, 2)
WHERE department = 'Computer Science' AND gpa < 9.9;

-- Update with a subquery
UPDATE students
SET gpa = (SELECT AVG(gpa) FROM students WHERE department = 'Computer Science')
WHERE student_id = 100;  -- set one student's GPA to dept average

-- Update with CASE (conditional update)
UPDATE students
SET department = CASE
    WHEN department = 'CS' THEN 'Computer Science'
    WHEN department = 'ECE' THEN 'Electronics'
    WHEN department = 'ME'  THEN 'Mechanical'
    ELSE department
END
WHERE department IN ('CS', 'ECE', 'ME');
SQL – DELETE and TRUNCATE
-- Delete a specific row
DELETE FROM students WHERE student_id = 5;

-- Delete with condition
DELETE FROM students WHERE gpa < 5.0 AND enrollment_date < '2020-01-01';

-- Delete using subquery (delete students not enrolled in any course)
DELETE FROM students
WHERE student_id NOT IN (
    SELECT DISTINCT student_id FROM enrollments
);

-- Safe delete: check count before deleting
SELECT COUNT(*) FROM students WHERE age > 60;  -- verify first
DELETE FROM students WHERE age > 60;

-- TRUNCATE vs DELETE difference:
-- DELETE: row-by-row, triggers fire, can be rolled back, keeps AUTO_INCREMENT
-- TRUNCATE: faster, no triggers, cannot be rolled back in MySQL, resets AUTO_INCREMENT

TRUNCATE TABLE students;      -- faster, resets auto_increment
DELETE FROM students;         -- slower, keeps auto_increment value
Warning: Always run a SELECT with the same WHERE clause before UPDATE or DELETE to verify which rows will be affected. A missing WHERE clause in DELETE or UPDATE affects ALL rows.

Chapter 4: SELECT Queries

SQL – Basic SELECT, WHERE, ORDER BY, LIMIT
-- Retrieve all columns and rows
SELECT * FROM students;

-- Retrieve specific columns
SELECT first_name, last_name, gpa FROM students;

-- Column aliases
SELECT
    CONCAT(first_name, ' ', last_name) AS full_name,
    gpa AS grade_point,
    department AS dept
FROM students;

-- DISTINCT: remove duplicate values
SELECT DISTINCT department FROM students;
SELECT COUNT(DISTINCT department) AS num_depts FROM students;

-- ── WHERE clause conditions ──────────────────────────────────────────

-- Comparison operators
SELECT * FROM students WHERE gpa > 9.0;
SELECT * FROM students WHERE age <= 20;
SELECT * FROM students WHERE department = 'Computer Science';

-- Logical operators (AND, OR, NOT)
SELECT * FROM students
WHERE department = 'Computer Science' AND age < 21;

SELECT * FROM students
WHERE department = 'Computer Science' OR department = 'Electronics';

SELECT * FROM students
WHERE NOT (department = 'Mechanical');

-- BETWEEN (inclusive on both ends)
SELECT * FROM students WHERE gpa BETWEEN 8.0 AND 9.0;
SELECT * FROM students WHERE enrollment_date BETWEEN '2022-01-01' AND '2023-12-31';

-- IN and NOT IN
SELECT * FROM students
WHERE department IN ('Computer Science', 'Electronics', 'Mathematics');

SELECT * FROM students
WHERE student_id NOT IN (SELECT student_id FROM enrollments);

-- LIKE: pattern matching
-- % = any number of characters, _ = exactly one character
SELECT * FROM students WHERE first_name LIKE 'R%';        -- starts with R
SELECT * FROM students WHERE first_name LIKE '%a';        -- ends with a
SELECT * FROM students WHERE first_name LIKE '%ra%';      -- contains 'ra'
SELECT * FROM students WHERE first_name LIKE '_r%';       -- 2nd char is 'r'
SELECT * FROM students WHERE email LIKE '%@gmail.com';    -- gmail users

-- IS NULL and IS NOT NULL
SELECT * FROM students WHERE department IS NULL;
SELECT * FROM students WHERE department IS NOT NULL;
SELECT * FROM students WHERE phone IS NULL AND gpa > 8.0;

-- ORDER BY
SELECT * FROM students ORDER BY gpa DESC;               -- highest GPA first
SELECT * FROM students ORDER BY department ASC, gpa DESC;  -- multi-column sort
SELECT * FROM students ORDER BY LENGTH(last_name), last_name;  -- by length

-- LIMIT and OFFSET (pagination)
SELECT * FROM students ORDER BY gpa DESC LIMIT 5;            -- top 5
SELECT * FROM students ORDER BY gpa DESC LIMIT 5 OFFSET 5;  -- rows 6-10
SELECT * FROM students ORDER BY gpa DESC LIMIT 5, 5;        -- MySQL shorthand: LIMIT offset, count

-- RAND() — random sample
SELECT * FROM students ORDER BY RAND() LIMIT 3;   -- 3 random students

SQL Execution Order

SQL clauses execute in this order:
FROM → JOIN → WHERE → GROUP BY → HAVING → SELECT → DISTINCT → ORDER BY → LIMIT

This is why you cannot use a column alias defined in SELECT inside a WHERE clause — WHERE executes before SELECT.
SQL – String, Date, and Math Functions
-- String functions
SELECT UPPER(first_name), LOWER(last_name) FROM students;
SELECT CONCAT(first_name, ' ', last_name) AS full_name FROM students;
SELECT LENGTH(email) AS email_length FROM students;
SELECT SUBSTRING(email, 1, 5) AS email_prefix FROM students;  -- first 5 chars
SELECT TRIM('  hello  ');        -- 'hello'
SELECT REPLACE(email, '@', ' at ') FROM students;

-- Date functions
SELECT CURDATE(), CURTIME(), NOW();   -- current date, time, datetime
SELECT YEAR(enrollment_date) AS year FROM students;
SELECT DATEDIFF(CURDATE(), enrollment_date) AS days_enrolled FROM students;
SELECT DATE_FORMAT(enrollment_date, '%d %M %Y') AS formatted FROM students;
SELECT DATE_ADD(enrollment_date, INTERVAL 1 YEAR) AS one_year_later FROM students;

-- Math functions
SELECT ABS(-15.7);          -- 15.7
SELECT CEIL(4.3);           -- 5
SELECT FLOOR(4.9);          -- 4
SELECT ROUND(9.567, 2);     -- 9.57
SELECT POWER(2, 10);        -- 1024
SELECT SQRT(144);           -- 12
SELECT MOD(17, 5);          -- 2 (remainder)

-- Conditional: COALESCE (return first non-NULL)
SELECT COALESCE(phone, 'No phone') AS contact FROM students;
-- NULLIF(a, b): returns NULL if a = b, else returns a
SELECT NULLIF(department, 'Unknown') FROM students;

Chapter 5: Aggregate Functions and GROUP BY / HAVING

SQL – Aggregate Functions
-- COUNT: count rows
SELECT COUNT(*)           AS total_rows     FROM students;  -- all rows
SELECT COUNT(phone)       AS has_phone      FROM students;  -- non-NULL only
SELECT COUNT(DISTINCT department) AS dept_count FROM students;

-- SUM, AVG, MIN, MAX
SELECT
    SUM(gpa)        AS sum_gpa,
    AVG(gpa)        AS avg_gpa,
    MIN(gpa)        AS min_gpa,
    MAX(gpa)        AS max_gpa,
    ROUND(AVG(gpa), 2) AS rounded_avg
FROM students;

-- Aggregate on filtered rows
SELECT AVG(gpa) AS avg_cs_gpa
FROM students
WHERE department = 'Computer Science';

-- GROUP_CONCAT: aggregate strings (MySQL)
SELECT department, GROUP_CONCAT(first_name ORDER BY first_name SEPARATOR ', ') AS names
FROM students
GROUP BY department;
SQL – GROUP BY and HAVING
-- Basic GROUP BY: one row per group
SELECT
    department,
    COUNT(*)          AS student_count,
    ROUND(AVG(gpa),2) AS avg_gpa,
    MAX(gpa)          AS top_gpa,
    MIN(gpa)          AS bottom_gpa
FROM students
GROUP BY department
ORDER BY avg_gpa DESC;

-- GROUP BY multiple columns
SELECT department, YEAR(enrollment_date) AS year, COUNT(*) AS count
FROM students
GROUP BY department, YEAR(enrollment_date)
ORDER BY department, year;

-- HAVING: filter AFTER grouping (like WHERE but for groups)
SELECT department, COUNT(*) AS count, ROUND(AVG(gpa), 2) AS avg
FROM students
GROUP BY department
HAVING COUNT(*) >= 3 AND AVG(gpa) > 8.0;

-- WHERE vs HAVING
-- WHERE filters individual rows BEFORE grouping
-- HAVING filters groups AFTER grouping

-- Combining WHERE + GROUP BY + HAVING + ORDER BY
SELECT
    department,
    COUNT(*)            AS enrolled_count,
    ROUND(AVG(gpa), 2)  AS avg_gpa
FROM students
WHERE enrollment_date >= '2022-01-01'    -- filter rows first
GROUP BY department
HAVING COUNT(*) > 2                      -- filter groups
ORDER BY avg_gpa DESC                    -- sort result
LIMIT 5;                                 -- take top 5

-- ROLLUP: add subtotals and grand total
SELECT department, YEAR(enrollment_date) AS year, COUNT(*) AS count
FROM students
GROUP BY department, YEAR(enrollment_date) WITH ROLLUP;

Chapter 6: JOINs — Complete Guide

⚫

INNER JOIN

Only rows that match in BOTH tables

⚬

LEFT JOIN

All rows from LEFT + matching from right (NULL if no match)

⚭

RIGHT JOIN

All rows from RIGHT + matching from left (NULL if no match)

⚮

FULL OUTER

All rows from BOTH tables (NULL where no match)

⟳

SELF JOIN

Table joined with itself using aliases

⨯

CROSS JOIN

Cartesian product: every row with every other row

SQL – INNER JOIN
-- INNER JOIN: only students who are enrolled in at least one course
SELECT
    s.first_name,
    s.last_name,
    c.course_name,
    e.grade,
    e.semester
FROM students s
INNER JOIN enrollments e ON s.student_id = e.student_id
INNER JOIN courses c     ON e.course_id  = c.course_id
ORDER BY s.last_name, c.course_name;

-- JOIN with USING shorthand (when column names are identical)
SELECT s.first_name, c.course_name
FROM students s
JOIN enrollments e USING (student_id)
JOIN courses c     USING (course_id);

-- Filtering in JOINs
SELECT s.first_name, c.course_name, e.grade
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c     ON e.course_id  = c.course_id
WHERE s.department = 'Computer Science'
  AND c.credits >= 3
  AND e.grade IN ('A', 'A+', 'O');

-- Aggregate over join
SELECT
    s.department,
    COUNT(DISTINCT s.student_id) AS enrolled_students,
    COUNT(e.enrollment_id)       AS total_enrollments,
    ROUND(AVG(c.credits), 1)     AS avg_credits_per_enroll
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c     ON e.course_id  = c.course_id
GROUP BY s.department
ORDER BY total_enrollments DESC;
SQL – LEFT JOIN, RIGHT JOIN, FULL OUTER JOIN
-- LEFT JOIN: show ALL students, even if not enrolled in any course
SELECT
    s.first_name,
    s.last_name,
    COUNT(e.enrollment_id) AS courses_enrolled
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
GROUP BY s.student_id, s.first_name, s.last_name
ORDER BY courses_enrolled DESC;

-- Find students NOT enrolled in any course
SELECT s.first_name, s.last_name
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
WHERE e.student_id IS NULL;   -- NULL means no match in right table

-- RIGHT JOIN: all courses, including those with no students
SELECT
    c.course_name,
    COUNT(e.student_id) AS enrolled_count
FROM students s
RIGHT JOIN enrollments e ON s.student_id = e.student_id
RIGHT JOIN courses c     ON e.course_id  = c.course_id
GROUP BY c.course_id, c.course_name;

-- FULL OUTER JOIN (PostgreSQL / MS SQL)
SELECT s.first_name, c.course_name
FROM students s
FULL OUTER JOIN enrollments e ON s.student_id = e.student_id
FULL OUTER JOIN courses c     ON e.course_id  = c.course_id;

-- MySQL: simulate FULL OUTER JOIN using UNION
SELECT s.first_name, c.course_name
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
LEFT JOIN courses c     ON e.course_id  = c.course_id
UNION
SELECT s.first_name, c.course_name
FROM students s
RIGHT JOIN enrollments e ON s.student_id = e.student_id
RIGHT JOIN courses c     ON e.course_id  = c.course_id;
SQL – SELF JOIN and CROSS JOIN
-- SELF JOIN: find pairs of students in the same department
SELECT
    a.first_name AS student1,
    b.first_name AS student2,
    a.department
FROM students a
INNER JOIN students b
    ON a.department = b.department
    AND a.student_id < b.student_id   -- avoid (A,B) and (B,A) duplicates
ORDER BY a.department, a.first_name;

-- SELF JOIN: find students with GPA higher than their department average
SELECT
    s1.first_name,
    s1.department,
    s1.gpa,
    ROUND(s2_avg.dept_avg, 2) AS dept_avg
FROM students s1
JOIN (
    SELECT department, AVG(gpa) AS dept_avg
    FROM students
    GROUP BY department
) AS s2_avg ON s1.department = s2_avg.department
WHERE s1.gpa > s2_avg.dept_avg
ORDER BY s1.department, s1.gpa DESC;

-- CROSS JOIN: generate all (student, course) combinations for timetabling
SELECT s.first_name, c.course_name
FROM students s
CROSS JOIN courses c
WHERE s.department = c.department
ORDER BY s.first_name, c.course_name;
-- Warning: produces m × n rows — use carefully!

Chapter 7: Subqueries and Correlated Queries

SQL – Subqueries in WHERE, SELECT, FROM
-- Subquery in WHERE: students with above-average GPA
SELECT first_name, gpa
FROM students
WHERE gpa > (SELECT AVG(gpa) FROM students);

-- Subquery with IN: students enrolled in 4-credit courses
SELECT first_name FROM students
WHERE student_id IN (
    SELECT student_id FROM enrollments
    WHERE course_id IN (
        SELECT course_id FROM courses WHERE credits = 4
    )
);

-- Subquery with NOT IN: students not enrolled in any course
SELECT first_name, last_name FROM students
WHERE student_id NOT IN (
    SELECT DISTINCT student_id FROM enrollments
);

-- Subquery in SELECT (scalar subquery — must return exactly one row/column)
SELECT
    first_name,
    gpa,
    (SELECT AVG(gpa) FROM students) AS overall_avg,
    gpa - (SELECT AVG(gpa) FROM students) AS diff_from_avg
FROM students;

-- Subquery in FROM (derived table / inline view)
SELECT dept_stats.department, dept_stats.avg_gpa, dept_stats.count
FROM (
    SELECT
        department,
        ROUND(AVG(gpa), 2) AS avg_gpa,
        COUNT(*)           AS count
    FROM students
    GROUP BY department
) AS dept_stats
WHERE dept_stats.count >= 3
ORDER BY dept_stats.avg_gpa DESC;
SQL – Correlated Subquery and EXISTS
-- Correlated subquery: references outer query (runs once per outer row)
-- Find the top student in each department
SELECT s.first_name, s.department, s.gpa
FROM students s
WHERE s.gpa = (
    SELECT MAX(gpa)
    FROM students s2
    WHERE s2.department = s.department  -- correlates with outer query
);

-- EXISTS: often faster than IN for large datasets
-- Students who are enrolled in at least one course
SELECT s.first_name, s.last_name
FROM students s
WHERE EXISTS (
    SELECT 1
    FROM enrollments e
    WHERE e.student_id = s.student_id
);

-- NOT EXISTS: students who have NOT submitted grades
SELECT c.course_name
FROM courses c
WHERE NOT EXISTS (
    SELECT 1 FROM enrollments e
    WHERE e.course_id = c.course_id AND e.grade IS NOT NULL
);

-- ANY / ALL operators
-- Students whose GPA is higher than ANY CS student
SELECT first_name, gpa FROM students
WHERE gpa > ANY (
    SELECT gpa FROM students WHERE department = 'Computer Science'
);

-- Students whose GPA is higher than ALL ME students
SELECT first_name, gpa FROM students
WHERE gpa > ALL (
    SELECT gpa FROM students WHERE department = 'Mechanical'
);

Chapter 8: Advanced SQL

Window Functions

Window functions perform calculations across a set of rows related to the current row — without collapsing them into groups like GROUP BY. They use the OVER() clause.
SQL – Window Functions: RANK, ROW_NUMBER, LAG, LEAD, Running Total
-- RANK vs DENSE_RANK vs ROW_NUMBER
SELECT
    first_name,
    department,
    gpa,
    RANK()        OVER (ORDER BY gpa DESC) AS overall_rank,
    DENSE_RANK()  OVER (ORDER BY gpa DESC) AS dense_rank,
    ROW_NUMBER()  OVER (ORDER BY gpa DESC) AS row_num
FROM students;
-- RANK:       gaps after ties (1,1,3,4)
-- DENSE_RANK: no gaps       (1,1,2,3)
-- ROW_NUMBER: always unique (1,2,3,4)

-- PARTITION BY: rank within each department
SELECT
    first_name,
    department,
    gpa,
    RANK() OVER (PARTITION BY department ORDER BY gpa DESC) AS dept_rank
FROM students;

-- LAG and LEAD: access adjacent rows
SELECT
    first_name,
    department,
    gpa,
    LAG(gpa, 1)  OVER (PARTITION BY department ORDER BY gpa DESC) AS prev_gpa,
    LEAD(gpa, 1) OVER (PARTITION BY department ORDER BY gpa DESC) AS next_gpa,
    gpa - LAG(gpa, 1) OVER (PARTITION BY department ORDER BY gpa DESC) AS gpa_diff
FROM students;

-- Running total and moving average
SELECT
    student_id,
    first_name,
    gpa,
    SUM(gpa)  OVER (ORDER BY student_id ROWS UNBOUNDED PRECEDING) AS running_sum,
    AVG(gpa)  OVER (ORDER BY student_id ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) AS moving_avg_3
FROM students;

-- NTILE: divide into quartiles
SELECT
    first_name,
    gpa,
    NTILE(4) OVER (ORDER BY gpa DESC) AS quartile
FROM students;
-- 1 = top 25%, 4 = bottom 25%

-- FIRST_VALUE and LAST_VALUE
SELECT
    first_name,
    department,
    gpa,
    FIRST_VALUE(first_name) OVER (PARTITION BY department ORDER BY gpa DESC) AS top_student,
    MAX(gpa)                OVER (PARTITION BY department) AS dept_max_gpa
FROM students;

Common Table Expressions (CTEs)

SQL – CTEs and Recursive CTEs
-- Basic CTE (WITH clause)
WITH dept_stats AS (
    SELECT
        department,
        COUNT(*)            AS student_count,
        ROUND(AVG(gpa), 2)  AS avg_gpa
    FROM students
    GROUP BY department
)
SELECT *
FROM dept_stats
WHERE avg_gpa > 8.0
ORDER BY avg_gpa DESC;

-- Multiple CTEs chained together
WITH
dept_averages AS (
    SELECT department, AVG(gpa) AS avg_gpa
    FROM students
    GROUP BY department
),
above_avg_students AS (
    SELECT s.student_id, s.first_name, s.department, s.gpa, da.avg_gpa
    FROM students s
    JOIN dept_averages da ON s.department = da.department
    WHERE s.gpa > da.avg_gpa
)
SELECT
    a.first_name,
    a.department,
    ROUND(a.gpa, 2) AS gpa,
    ROUND(a.avg_gpa, 2) AS dept_avg
FROM above_avg_students a
ORDER BY a.department, a.gpa DESC;

-- Recursive CTE: generate a sequence of numbers 1-10
WITH RECURSIVE numbers AS (
    SELECT 1 AS n          -- anchor member
    UNION ALL
    SELECT n + 1           -- recursive member
    FROM numbers
    WHERE n < 10
)
SELECT * FROM numbers;

-- Recursive CTE: organizational hierarchy
WITH RECURSIVE org_chart AS (
    -- anchor: top-level employees (no manager)
    SELECT employee_id, name, manager_id, 1 AS level
    FROM employees
    WHERE manager_id IS NULL
    UNION ALL
    -- recursive: employees reporting to someone
    SELECT e.employee_id, e.name, e.manager_id, oc.level + 1
    FROM employees e
    JOIN org_chart oc ON e.manager_id = oc.employee_id
)
SELECT REPEAT('  ', level-1) || name AS hierarchy, level
FROM org_chart
ORDER BY level, name;

CASE Expressions

SQL – CASE Expression
-- Searched CASE (most common)
SELECT
    first_name,
    gpa,
    CASE
        WHEN gpa >= 9.5 THEN 'Distinction'
        WHEN gpa >= 9.0 THEN 'First Class'
        WHEN gpa >= 7.5 THEN 'Second Class'
        WHEN gpa >= 6.0 THEN 'Pass'
        ELSE 'Fail'
    END AS result
FROM students;

-- Simple CASE
SELECT
    first_name,
    department,
    CASE department
        WHEN 'Computer Science' THEN 'CS'
        WHEN 'Electronics'      THEN 'ECE'
        WHEN 'Mechanical'       THEN 'ME'
        ELSE 'Other'
    END AS dept_code
FROM students;

-- CASE in aggregate (pivot-like query)
SELECT
    department,
    SUM(CASE WHEN gpa >= 9.0 THEN 1 ELSE 0 END) AS distinction_count,
    SUM(CASE WHEN gpa >= 7.5 AND gpa < 9.0 THEN 1 ELSE 0 END) AS first_class_count,
    SUM(CASE WHEN gpa < 7.5 THEN 1 ELSE 0 END) AS others_count
FROM students
GROUP BY department;

-- CASE in ORDER BY
SELECT first_name, department
FROM students
ORDER BY
    CASE department
        WHEN 'Computer Science' THEN 1
        WHEN 'Electronics'      THEN 2
        ELSE 3
    END,
    first_name;

Chapter 9: Indexes and Query Performance

SQL – Creating and Using Indexes
-- Single column index
CREATE INDEX idx_department ON students(department);
CREATE INDEX idx_gpa ON students(gpa);

-- Unique index (also enforces uniqueness constraint)
CREATE UNIQUE INDEX idx_email ON students(email);

-- Composite index (multiple columns — column order matters!)
-- Most selective column first; covers queries filtering on (department) OR (department, gpa)
CREATE INDEX idx_dept_gpa ON students(department, gpa DESC);

-- Full-text index (for text search — MySQL)
CREATE FULLTEXT INDEX idx_ft_name ON students(first_name, last_name);
SELECT * FROM students WHERE MATCH(first_name, last_name) AGAINST ('rahul sharma' IN NATURAL LANGUAGE MODE);

-- Drop an index
DROP INDEX idx_gpa ON students;          -- MySQL
DROP INDEX idx_gpa;                       -- PostgreSQL (use ALTER TABLE)

-- Show all indexes on a table
SHOW INDEXES FROM students;              -- MySQL
SELECT * FROM pg_indexes WHERE tablename = 'students';  -- PostgreSQL
SQL – EXPLAIN to Analyze Query Plans
-- EXPLAIN: shows how MySQL executes a query
EXPLAIN SELECT * FROM students WHERE department = 'Computer Science';
-- Look for: type (ALL = full scan, ref/range = index used), rows, Extra

-- EXPLAIN ANALYZE (PostgreSQL — actually executes the query)
EXPLAIN ANALYZE SELECT * FROM students WHERE department = 'Computer Science';

-- Index is NOT used when:
-- 1. Column is inside a function: WHERE YEAR(enrollment_date) = 2023
--    FIX: WHERE enrollment_date BETWEEN '2023-01-01' AND '2023-12-31'
-- 2. Using != or NOT IN (usually)
-- 3. Leading wildcard LIKE: WHERE name LIKE '%rahul' (no index; '%rahul%' either)
--    FIX: WHERE name LIKE 'rahul%'  (index IS used for trailing wildcard)
-- 4. Implicit type conversion: WHERE student_id = '123' (string vs int)

-- Covering index: all columns needed are IN the index
-- Query: SELECT gpa FROM students WHERE department = 'CS'
-- Index on (department, gpa) covers this query — no table lookup needed!
CREATE INDEX idx_dept_covering ON students(department, gpa, first_name);

15 SQL Interview Questions

1. Find the second highest GPA in the students table

Easy
SQL
-- Method 1: LIMIT + OFFSET
SELECT DISTINCT gpa FROM students ORDER BY gpa DESC LIMIT 1 OFFSET 1;

-- Method 2: Subquery
SELECT MAX(gpa) AS second_highest
FROM students
WHERE gpa < (SELECT MAX(gpa) FROM students);

-- Method 3: Dense rank (handles NULLs and ties correctly)
SELECT gpa FROM (
    SELECT gpa, DENSE_RANK() OVER (ORDER BY gpa DESC) AS rnk
    FROM students
) ranked
WHERE rnk = 2
LIMIT 1;

2. Find duplicate emails in the students table

Easy
SQL
SELECT email, COUNT(*) AS occurrences
FROM students
GROUP BY email
HAVING COUNT(*) > 1
ORDER BY occurrences DESC;

3. Delete duplicate rows, keeping only the one with the lowest student_id

Medium
SQL – MySQL
DELETE FROM students
WHERE student_id NOT IN (
    SELECT MIN(student_id)
    FROM students
    GROUP BY email
);

4. Get the department-wise top student (highest GPA) for each department

Medium
SQL – Window Function
SELECT first_name, department, gpa
FROM (
    SELECT first_name, department, gpa,
           RANK() OVER (PARTITION BY department ORDER BY gpa DESC) AS rnk
    FROM students
) ranked
WHERE rnk = 1;

5. Find students who enrolled in ALL available courses

Medium
SQL
SELECT student_id, COUNT(DISTINCT course_id) AS courses_taken
FROM enrollments
GROUP BY student_id
HAVING COUNT(DISTINCT course_id) = (SELECT COUNT(*) FROM courses);

6. Pivot: Show count of students per department as columns

Medium
SQL – Conditional Aggregation
SELECT
    SUM(CASE WHEN department = 'Computer Science' THEN 1 ELSE 0 END) AS CS,
    SUM(CASE WHEN department = 'Electronics'       THEN 1 ELSE 0 END) AS ECE,
    SUM(CASE WHEN department = 'Mechanical'        THEN 1 ELSE 0 END) AS ME,
    SUM(CASE WHEN department = 'Mathematics'       THEN 1 ELSE 0 END) AS Math
FROM students;

7. Find the running total of enrollments per day

Medium
SQL – Window Function
SELECT
    DATE(enrolled_at) AS enroll_date,
    COUNT(*) AS daily_count,
    SUM(COUNT(*)) OVER (ORDER BY DATE(enrolled_at)) AS running_total
FROM enrollments
GROUP BY DATE(enrolled_at)
ORDER BY enroll_date;

8. Find courses with no enrolled students

Easy
SQL
-- Method 1: LEFT JOIN + NULL check
SELECT c.course_name
FROM courses c
LEFT JOIN enrollments e ON c.course_id = e.course_id
WHERE e.course_id IS NULL;

-- Method 2: NOT IN
SELECT course_name FROM courses
WHERE course_id NOT IN (SELECT DISTINCT course_id FROM enrollments);

-- Method 3: NOT EXISTS
SELECT course_name FROM courses c
WHERE NOT EXISTS (SELECT 1 FROM enrollments e WHERE e.course_id = c.course_id);

9. Update salary (or GPA) by rank — top 10% get +10%, bottom 10% get -5%

Hard
SQL
WITH ranked AS (
    SELECT student_id,
           NTILE(10) OVER (ORDER BY gpa DESC) AS decile
    FROM students
)
UPDATE students s
JOIN ranked r ON s.student_id = r.student_id
SET s.gpa = CASE
    WHEN r.decile = 1  THEN LEAST(s.gpa * 1.10, 10.0)   -- top 10%
    WHEN r.decile = 10 THEN GREATEST(s.gpa * 0.95, 0.0)  -- bottom 10%
    ELSE s.gpa
END;
